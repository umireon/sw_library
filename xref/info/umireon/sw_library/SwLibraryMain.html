25  package info.umireon.sw_library;
26  
27  import java.io.IOException;
28  import java.io.InputStream;
29  import java.io.PrintStream;
30  
31  /**
32   * 図書館ドメインの実装のメインクラスです.
33   * @author Kaito Udagawa
34   */
35  public final class SwLibraryMain {
36  
37      /**
38       * 図書館ドメインの実装のエントリポイントです.
39       * @param args the command line arguments
40       * @throws java.io.IOException 入出力エラーが発生した場合
41       */
42      public static void main(final String[] args) throws IOException {
43          SwLibraryMain main;
44          main = new SwLibraryMain(System.in, System.out);
45          main.start(args);
46      }
47  
48      /**
49       * 標準入力です.
50       */
51      private final transient InputStream stdin;
52  
53      /**
54       * 標準出力です.
55       */
56      private final transient PrintStream stdout;
57  
58      /**
59       * 図書館ドメインの実装のインスタンスを作成します.
60       * @param sin 標準入力
61       * @param sout 標準出力
62       */
63      public SwLibraryMain(final InputStream sin, final PrintStream sout) {
64          this.stdin = sin;
65          this.stdout = sout;
66      }
67  
68      /**
69       * 図書館ドメインの実装を開始します.
70       * @param args 引数
71       * @throws IOException 入出力エラーが発生した場合
72       */
73      public void start(final String[] args) throws IOException {
74          EntityBook book1, book2;
75          EntityMagazine mag1;
76          EntityUser user1, user2;
77          EntityReservation res1;
78  
79          user1 = new EntityUser("user1");
80          user2 = new EntityUser("user2");
81  
82          res1 = new EntityReservation(user1);
83  
84          book1 = new EntityBook("book1");
85          book2 = new EntityBook("book2");
86          book2.setStatus(new EntityLoan(user2, new EntityDate()));
87          mag1 = new EntityMagazine("mag1");
88          mag1.addReservation(res1);
89  
90          ControlMaterial ctrlMaterial = new ControlMaterial();
91          ctrlMaterial.addMaterial(book1);
92          ctrlMaterial.addMaterial(book2);
93          ctrlMaterial.addMaterial(mag1);
94          ControlUser ctrlUser = new ControlUser();
95          ctrlUser.addUser(user1);
96          ctrlUser.addUser(user2);
97          BoundaryLend boundLend = new BoundaryLend(ctrlMaterial, ctrlUser,
98                  stdin, stdout);
99  
100         boundLend.listen();
101     }
102 }
